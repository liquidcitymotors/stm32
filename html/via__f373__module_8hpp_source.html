137  inline void setLogicOutputsLEDOn(uint32_t logicA, uint32_t auxLogic,
138  uint32_t shA, uint32_t shB) {
139 
140  // LEDA_HIGH_MASK -> SH_A_SAMPLE_MASK >> 16 >> 1 (pin 8 to pin 7, F)
141  // LEDB_HIGH_MASK -> SH_B_SAMPLE_MASK >> 16 << 5 (pin 9 to pin 14, C)
142  // LEDC_HIGH_MASK -> ALOGIC_HIGH_MASK >> 16 >> 11 (pin 13 to pin 2, A)
143  // LEDD_HIGH_MASK -> BLOGIC_HIGH_MASK >> 16 >> 13 (pin 15 to pin 2, B)
144 
145  #define LEDA_MASK (__ROR(shA, 16) >> 1)
146  #define LEDB_MASK (__ROR(shB, 16) << 5)
147  #define LEDC_MASK (__ROR(logicA, 16) >> 11)
148 
149  //combine the mask variables for a shared GPIO group with a bitwise or
150  *aLogicOutput = (logicA | LEDB_MASK);
151 
152  *auxLogicOutput = (auxLogic | LEDC_MASK);
153 
154  *shAOutput = (shA | shB);
155 
156  *ledAOutput = LEDA_MASK;
157 
158  }
